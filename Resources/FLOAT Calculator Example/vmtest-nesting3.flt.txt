float x;
float y;
{
  x = 2;
  write x;      # writes 2.000000
  y = x + 7;
  write y;      # writes 9.000000
  if (x <= y) {
     write y + 112;  # writes 121.000000
     if (x + 0 < y - 20) {
        write y + 1;
     }
  }
}
