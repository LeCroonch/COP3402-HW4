bool b, c;
float x, y, nl;
{
   x = 10;
   y = 121;
   b = x < y;
   {
      write y;   # writes 121.000000
      nl = x - 0;
      write nl;  # writes 10.000000
   }
   c = !b;
   if (b != c) {
      write y - 1;   # writes 120.000000
      write nl * nl; # writes 100.000000
   }
}
